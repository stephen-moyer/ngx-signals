createAutocomplete

Usage

General usage

Creates an autocomplete object that can be used to load suggestions for an input 

    
      readonly autocomplete = createAutocomplete(
        (value: string) => of(filterStates(value)).pipe(delay(150)),
        {{ '{' }}
          // how many ms before calling the function to get the suggestions observable
          // defaults to 300 ms
          delay: 300,
        {{ '}' }}
      );

      autocomplete.search('mar')

suggestions: {{ autocomplete().value | json }}
